  	maxlines = 10 # if we read ten lines with seeing "^---", something's fucky
  	header = list()
  	with open(filename) as f:

  		if YAML_DELIM.match(line):

  		    line = f.next()
  		    try:
  			while YAML_DELIM.match(line) is None:
  			    header.append(line.strip())
  			    line = f.next()
  		    except StopIteration:
  			msg = "file '%s' has an unclosed YAML delimeter"
  			raise EnvironmentError(1, msg % filename, filename)
  		if len(header) > 0:
  		    # we've loaded all the YAML
  		    break
  	yaml_header = yaml.load('\n'.join(header))
  	if 'body' in yaml_header:
  	    msg = "YAML header cannot contain special keyword 'body'"
  	    raise ValueError(msg)
